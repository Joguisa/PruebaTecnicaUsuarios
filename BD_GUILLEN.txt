CREATE DATABASE BD_GUILLEN

USE BD_GUILLEN

CREATE TABLE Perfiles (
    PerfilID INT PRIMARY KEY IDENTITY(1,1),
    Nombre NVARCHAR(50) NOT NULL,
    Descripcion NVARCHAR(100) NOT NULL
);

CREATE TABLE Cargo (
    CargoID INT PRIMARY KEY IDENTITY(1,1),
    Nombre NVARCHAR(50) NOT NULL,
    Descripcion NVARCHAR(100) NOT NULL
);

CREATE TABLE Usuarios (
    UsuarioID INT PRIMARY KEY IDENTITY(1,1),
    Nombre NVARCHAR(50) NOT NULL,
    Apellido NVARCHAR(50) NOT NULL,
    Edad INT NOT NULL,
    FechaNacimiento DATE NOT NULL,
    Direccion NVARCHAR(100) NOT NULL,
    Estado INT NOT NULL DEFAULT 1,
    PerfilID INT FOREIGN KEY REFERENCES Perfiles(PerfilID),
    CargoID INT FOREIGN KEY REFERENCES Cargo(CargoID)
);
